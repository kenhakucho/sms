CREATE DATABASE smsdb DEFAULT CHARACTER SET utf8 COLLATE utf8_general_ci;

DROP TABLE user;
DROP TABLE room;
DROP TABLE member;
DROP TABLE message;
DROP TABLE stamp;
DROP TABLE item;

CREATE TABLE user(
id int not null auto_increment, 
mail varchar(255), 
password varchar(255), 
name varchar(40), 
nickname varchar(30), 
icon blob,
enable boolean default 1,
made timestamp,
PRIMARY KEY(id)
);
ALTER TABLE user ADD INDEX user_index_01(mail);

CREATE PROCEDURE user_id()
SELECT IFNULL(MAX(id)+1,0) FROM user;

CREATE TABLE room(
id int not null auto_increment, 
user_id int not null,
name varchar(20), 
icon varchar(255),
enable boolean default 1,
made timestamp,
PRIMARY KEY(id)
);

CREATE TABLE member(
room_id int not null, 
user_id int not null,
enable boolean default 1,
made timestamp,
PRIMARY KEY(room_id, user_id)
);
ALTER TABLE member ADD INDEX member_index_01(room_id);
ALTER TABLE member ADD INDEX member_index_02(user_id);

CREATE PROCEDURE user_id()
SELECT IFNULL(MAX(id)+1,0) FROM user;

CREATE TABLE post(
id int not null auto_increment, 
room_id int not null,
user_id int not null,
isstamp boolean not null default false,
message varchar(255), 
file varchar(255),
enable boolean default 1,
made timestamp,
PRIMARY KEY(id)
);
ALTER TABLE post ADD INDEX post_index_01(made);

CREATE TABLE stamp(
id int not null auto_increment, 
name varchar(100),
file varchar(255) not null,
enable boolean,
made timestamp,
PRIMARY KEY(id)
);

CREATE TABLE item(
id int not null auto_increment, 
user_id int not null,
stamp_id int not null,
file varchar(255) not null,
enable boolean default 1,
made timestamp,
PRIMARY KEY(id)
);
ALTER TABLE item ADD INDEX item_index_01(user_id);
